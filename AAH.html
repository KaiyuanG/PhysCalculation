AAH Model¶

The AAH Model is given by:

\begin{equation} \mathcal{H}_S = \Sigma_{r=1}^{N-1} (\hat{a}_r^\dagger \hat{a}_{r+1}+h.c.) + \Sigma_{r=1}^N 2\lambda \cos(2\pi r\alpha + \phi_0)\hat{a}_r^\dagger \hat{a}_r \end{equation}

We set $\lambda =1$, which is the critical point. When $\lambda < 1$, the energy eigenstates are declocalized; and when $\lambda > 1$, the energy eigenstates are localized.

import numpy as np
import math
import matplotlib.pyplot as plt
from scipy.linalg import eigh_tridiagonal

n = 400 # N
m = 400 # number of alpha

w = np.zeros((n,m))

a = np.linspace(0,1,m,endpoint=True) # a = p/q = p/399, p ranging from 0 to 399

for j in range (m):
    d = np.zeros(n)
    for i in range(n):
        d[i] = math.cos(2 * math.pi * i * a[j])

    e = np.ones(n-1)
    w[:,j], v = eigh_tridiagonal(d, e)

plt.figure(figsize=(10,10))  
for i in range(n):
    plt.scatter(a, w[i,:],s=3.,c='gray')

plt.xlabel("")
plt.ylabel("Energy")
plt.show()

n = 400 # N
m = 400 # number of phi

w = np.zeros((n,m))

p = np.linspace(-math.pi,math.pi,m,endpoint=True) # phi ranging from -pi to pi

for j in range (m):
    d = np.zeros(n)
    for i in range(n):
        d[i] = math.cos(2 * math.pi * i /4 + p[j])

    e = np.ones(n-1)
    w[:,j], v = eigh_tridiagonal(d, e)

plt.figure(figsize=(10,8))  
for i in range(n):
    plt.scatter(p/math.pi, w[i,:],s=3.,c='gray')

plt.xlabel(" ()")
plt.ylabel("Energy")
plt.show()
